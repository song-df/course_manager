# 支付宝电脑网站支付 — 个体工商户接入实施方案

> **适用对象**：个体工商户（已有营业执照）
> **产品**：支付宝电脑网站支付（`alipay.trade.page.pay`）
> **费率**：0.6%/笔 | **沙箱环境**：免费调试 | **Python SDK**：alipay-sdk-python

---

## 一、前置准备（人工操作，约 1-2 天）

### 1.1 注册企业支付宝账号
1. 访问 [b.alipay.com](https://b.alipay.com/)，用个体工商户营业执照注册
2. 完成企业实名认证（需上传营业执照照片、法人身份证）
3. 确保支付宝账号主体名称与 ICP 备案主体一致（浙ICP备2022030063号-1）

### 1.2 入驻开放平台 & 创建应用
1. 访问 [open.alipay.com](https://open.alipay.com/)，用企业支付宝账号登录
2. 进入「控制台」→「创建应用」→ 选择「自研应用」
3. 应用类型：**网页/移动应用**
4. 配置应用：

| 配置项 | 值 |
|--------|-----|
| **接口加签方式** | RSA2（推荐）→ 在线生成密钥对 → 保存私钥 → 上传公钥 |
| **应用网关** | `https://aiotedu.cc` |
| **授权回调地址** | `https://aiotedu.cc/api/payment/return` |
| **应用功能** | 添加「电脑网站支付」能力 |

5. 记录以下关键参数：

| 参数 | 说明 | 示例 |
|------|------|------|
| **app_id** | 应用ID | `2024100123456789` |
| **应用私钥** | 自己生成的 RSA 私钥（PEM格式） | `MIIEvQIBADANBg...` |
| **支付宝公钥** | 上传公钥后支付宝返回的公钥 | `MIIBIjANBgkqh...` |

### 1.3 签约电脑网站支付产品
1. 在开放平台「产品管理」中找到「电脑网站支付」
2. 点击「签约」，填写个体户信息
3. 等待审核（通常 1 个工作日）
4. 审核通过后即可使用

### 1.4 商品内容配置
支付页面需要展示明确的商品信息。建议提前准备：

| 商品名称 | 内容 | 建议售价 |
|----------|------|----------|
| 单系列畅学卡 | 任选1个系列全部课程 | ¥99 |
| 季度全站通 | 3个月全站畅学 | ¥299 |
| 年度全站通 | 12个月全站畅学 | ¥699 |
| 永久会员 | 终身全站畅学 | ¥1,299 |

> 价格可根据实际调整。所有商品用同一个产品码 `FAST_INSTANT_TRADE_PAY`。

# ============================================================
# 支付宝电脑网站支付模块
# ============================================================
import hashlib
import sqlite3
import time
import uuid
import requests
from alipay import AliPay
from urllib.parse import parse_qs

# ---------- 配置 ----------
ALIPAY_CONFIG_FILE = os.path.join(os.path.dirname(os.path.abspath(__file__)), 'alipay_config.json')
ORDERS_DB = os.path.join(os.path.dirname(os.path.abspath(__file__)), 'orders.db')

# 加载支付宝配置
with open(ALIPAY_CONFIG_FILE, 'r') as f:
    alipay_cfg = json.load(f)

# 初始化支付宝 SDK
alipay = AliPay(
    appid=alipay_cfg['app_id'],
    app_notify_url=alipay_cfg['notify_url'],      # 异步通知地址
    app_private_key_string=alipay_cfg['app_private_key'],
    alipay_public_key_string=alipay_cfg['alipay_public_key'],
    sign_type='RSA2',
    debug=alipay_cfg.get('debug', False),          # True=沙箱, False=正式
)

# 定价方案
PRICE_PLANS = {
    'series_99':  {'name': '单系列畅学卡', 'price': 99.00,  'desc': '任选1个系列全部课程'},
    'quarter_299': {'name': '季度全站通',  'price': 299.00, 'desc': '3个月全站畅学'},
    'year_699':   {'name': '年度全站通',   'price': 699.00, 'desc': '12个月全站畅学'},
    'forever_1299': {'name': '永久会员',   'price': 1299.00, 'desc': '终身全站畅学'},
}

# ---------- 订单数据库 ----------
def init_orders_db():
    conn = sqlite3.connect(ORDERS_DB)
    conn.execute('''CREATE TABLE IF NOT EXISTS orders (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        out_trade_no TEXT UNIQUE NOT NULL,
        plan_id TEXT NOT NULL,
        total_amount REAL NOT NULL,
        trade_status TEXT DEFAULT 'WAIT_BUYER_PAY',
        trade_no TEXT,
        buyer_id TEXT,
        invite_code TEXT,
        created_at INTEGER NOT NULL,
        paid_at INTEGER
    )''')
    conn.commit()
    return conn

# ---------- API 端点处理 ----------
def handle_payment_create(handler):
    """POST /api/payment/create — 创建订单，返回跳转支付宝的 HTML form"""
    try:
        body_len = int(handler.headers.get('Content-Length', 0))
        body = json.loads(handler.rfile.read(body_len))
        plan_id = body.get('plan_id', 'quarter_299')
    except:
        handler.send_error(400)
        return

    plan = PRICE_PLANS.get(plan_id, PRICE_PLANS['quarter_299'])

    # 生成唯一商户订单号
    out_trade_no = f"ZL{int(time.time() * 1000)}{uuid.uuid4().hex[:6].upper()}"

    # 保存订单
    conn = init_orders_db()
    conn.execute(
        "INSERT INTO orders (out_trade_no, plan_id, total_amount, created_at) VALUES (?, ?, ?, ?)",
        (out_trade_no, plan_id, plan['price'], int(time.time()))
    )
    conn.commit()
    conn.close()

    # 调用支付宝生成支付表单
    order_string = alipay.api_alipay_trade_page_pay(
        out_trade_no=out_trade_no,
        total_amount=str(plan['price']),
        subject=f"智联学习云 - {plan['name']}",
        body=plan['desc'],
        product_code='FAST_INSTANT_TRADE_PAY',
        # return_url 仅用于同步跳转展示，不可信任其支付结果
        return_url=f"https://aiotedu.cc/api/payment/return?out_trade_no={out_trade_no}",
        timeout_express='15m',  # 15分钟未支付自动关闭
    )

    # 支付宝 SDK 返回的是完整 HTML form 字符串（含自动提交脚本）
    handler.send_response(200)
    handler.send_header('Content-Type', 'text/html; charset=utf-8')
    handler.end_headers()
    handler.wfile.write(order_string.encode('utf-8'))


def handle_payment_notify(handler):
    """POST /api/payment/notify — 接收支付宝异步通知，验签后发邀请码"""
    body_len = int(handler.headers.get('Content-Length', 0))
    body = handler.rfile.read(body_len).decode('utf-8')

    # 解析通知参数
    params = parse_qs(body)
    # parse_qs 会把值变成 list，转回单值 dict
    notify_data = {k: v[0] for k, v in params.items()}

    # 验签
    sign = notify_data.pop('sign', '')
    sign_type = notify_data.pop('sign_type', 'RSA2')
    verified = alipay.verify(notify_data, sign)

    if not verified:
        log_payment("异步通知验签失败")
        handler.send_response(400)
        handler.end_headers()
        handler.wfile.write(b'failure')
        return

    # 验签通过，校验关键参数
    out_trade_no = notify_data.get('out_trade_no', '')
    trade_no = notify_data.get('trade_no', '')
    trade_status = notify_data.get('trade_status', '')
    total_amount = float(notify_data.get('total_amount', 0))
    app_id = notify_data.get('app_id', '')
    seller_id = notify_data.get('seller_id', '')

    log_payment(f"通知: out_trade_no={out_trade_no}, status={trade_status}, amount={total_amount}")

    # 校验 app_id
    if app_id != alipay_cfg['app_id']:
        log_payment(f"app_id 不匹配: {app_id}")
        handler.send_response(200)
        handler.end_headers()
        handler.wfile.write(b'success')  # 返回 success 避免支付宝重试
        return

    conn = init_orders_db()
    order = conn.execute(
        "SELECT * FROM orders WHERE out_trade_no=? AND total_amount=?",
        (out_trade_no, total_amount)
    ).fetchone()

    if not order:
        log_payment(f"订单不存在或金额不匹配: {out_trade_no}")
        handler.send_response(200)
        handler.end_headers()
        handler.wfile.write(b'success')
        conn.close()
        return

    # 只有支付成功才处理
    if trade_status in ('TRADE_SUCCESS', 'TRADE_FINISHED'):
        # 检查是否已处理（防止重复通知）
        existing = conn.execute(
            "SELECT invite_code FROM orders WHERE out_trade_no=? AND invite_code IS NOT NULL",
            (out_trade_no,)
        ).fetchone()

        if existing:
            log_payment(f"订单已处理过: {out_trade_no}, 邀请码: {existing[0]}")
        else:
            # 调用认证后端自动生成邀请码
            invite_code = generate_invite_code(out_trade_no)
            if invite_code:
                # 更新订单
                conn.execute(
                    "UPDATE orders SET trade_status=?, trade_no=?, buyer_id=?, invite_code=?, paid_at=? WHERE out_trade_no=?",
                    (trade_status, trade_no, notify_data.get('buyer_id', ''), invite_code, int(time.time()), out_trade_no)
                )
                conn.commit()
                log_payment(f"支付成功: {out_trade_no}, 邀请码: {invite_code}")
            else:
                log_payment(f"生成邀请码失败: {out_trade_no}")
                # 紧急：记录到待处理表，人工介入
                conn.execute(
                    "UPDATE orders SET trade_status='PAID_NO_CODE' WHERE out_trade_no=?",
                    (out_trade_no,)
                )
                conn.commit()

    elif trade_status == 'TRADE_CLOSED':
        conn.execute(
            "UPDATE orders SET trade_status='TRADE_CLOSED' WHERE out_trade_no=?",
            (out_trade_no,)
        )
        conn.commit()

    conn.close()

    # 必须返回 "success"，否则支付宝会重复通知（最多7次：4m/10m/10m/1h/2h/6h/15h）
    handler.send_response(200)
    handler.end_headers()
    handler.wfile.write(b'success')


def handle_payment_return(handler):
    """GET /api/payment/return — 同步跳转，展示支付结果（展示性，不可信任）"""
    from urllib.parse import urlparse
    qs = parse_qs(urlparse(handler.path).query)
    out_trade_no = qs.get('out_trade_no', [''])[0]

    # 主动查询支付结果（不依赖同步返回参数）
    conn = init_orders_db()
    order = conn.execute(
        "SELECT * FROM orders WHERE out_trade_no=?",
        (out_trade_no,)
    ).fetchone()
    conn.close()

    # 返回支付结果页
    result_page = os.path.join(ROOT, 'payment_return.html')
    if os.path.exists(result_page):
        handler.send_response(200)
        handler.send_header('Content-Type', 'text/html; charset=utf-8')
        handler.end_headers()
        html = open(result_page, 'r', encoding='utf-8').read()
        # 注入订单数据
        if order:
            html = html.replace('__OUT_TRADE_NO__', order[1])
            html = html.replace('__PLAN_NAME__', PRICE_PLANS.get(order[2], {}).get('name', ''))
            html = html.replace('__AMOUNT__', str(order[3]))
            html = html.replace('__STATUS__', order[4] if order[4] else 'WAIT_BUYER_PAY')
            html = html.replace('__INVITE_CODE__', order[6] or '')
        handler.wfile.write(html.encode('utf-8'))
    else:
        handler.send_error(404)


def handle_payment_query(handler):
    """POST /api/payment/query — 前端轮询查询支付结果"""
    try:
        body_len = int(handler.headers.get('Content-Length', 0))
        body = json.loads(handler.rfile.read(body_len))
        out_trade_no = body.get('out_trade_no', '')
    except:
        handler.send_error(400)
        return

    conn = init_orders_db()
    order = conn.execute(
        "SELECT out_trade_no, plan_id, total_amount, trade_status, invite_code FROM orders WHERE out_trade_no=?",
        (out_trade_no,)
    ).fetchone()
    conn.close()

    if order:
        result = {
            'out_trade_no': order[0],
            'plan_name': PRICE_PLANS.get(order[1], {}).get('name', ''),
            'total_amount': order[2],
            'trade_status': order[3],
            'invite_code': order[4],
            'paid': order[3] in ('TRADE_SUCCESS', 'TRADE_FINISHED'),
        }
    else:
        # 本地无记录，调用支付宝查询
        result = alipay.api_alipay_trade_query(out_trade_no=out_trade_no)
        trade_status = result.get('trade_status', '')
        result = {
            'out_trade_no': out_trade_no,
            'trade_status': trade_status,
            'paid': trade_status in ('TRADE_SUCCESS', 'TRADE_FINISHED'),
        }

    handler.send_response(200)
    handler.send_header('Content-Type', 'application/json; charset=utf-8')
    handler.send_header('Access-Control-Allow-Origin', '*')
    handler.end_headers()
    handler.wfile.write(json.dumps(result, ensure_ascii=False).encode('utf-8'))


def generate_invite_code(out_trade_no):
    """调用认证后端 API 自动生成邀请码"""
    try:
        resp = requests.post(
            alipay_cfg['auth_api_url'] + '/admin/invite-codes/external',
            json={
                'system_secret': alipay_cfg['auth_system_secret'],
                'count': 1,
                'memo': f'支付宝支付: {out_trade_no}',
            },
            timeout=10,
        )
        data = resp.json()
        codes = data.get('codes', [])
        return codes[0] if codes else None
    except Exception as e:
        log_payment(f"调用认证后端生成邀请码失败: {e}")
        return None


def log_payment(msg):
    now = time.strftime('%Y-%m-%d %H:%M:%S')
    print(f"[{now}] [PAYMENT] {msg}")

# 在 CourseHandler 类中新增 do_POST 方法
def do_POST(self):
    if self.path == '/api/payment/create':
        handle_payment_create(self)
        return
    if self.path == '/api/payment/notify':
        handle_payment_notify(self)
        return
    if self.path == '/api/payment/query':
        handle_payment_query(self)
        return
    self.send_error(404)

# 在 do_GET 中加入（放在 API 路由部分）
if self.path.startswith('/api/payment/return'):
    handle_payment_return(self)
    return

## 五、异常处理

| 异常场景 | 处理方式 |
|----------|----------|
| 异步通知未收到 | 用户在 `payment_return.html` 轮询 `/api/payment/query`，后端调用支付宝查询接口补单 |
| 认证后端不可用 | 订单标记 `PAID_NO_CODE`，后台定时任务（cron）每分钟重试 |
| 支付宝通知延迟 | 前端轮询最长 60 秒，超时提示用户联系客服 |
| 用户关闭支付页面 | 重新打开课程页时检测是否有未支付订单，引导继续支付或取消 |
| 邀请码耗尽 | 监控剩余量，<10 个自动告警 |
